历史场景法（Historical Scenarios）

```python
import numpy as np
import pandas as pd

# 1. 模拟生成历史数据 (实际操作中请使用 yfinance 或本地数据库读取)
# 假设我们有 2020 年 3 月危机期间的资产价格数据
np.random.seed(42)
dates = pd.date_range(start="2020-03-01", end="2020-03-31", freq="B")
tickers = ["AAPL", "MSFT", "GOOGL", "AMZN"]

# 模拟危机期间暴跌并剧烈波动的价格
mock_data = {
    "AAPL": np.linspace(75, 55, len(dates)) + np.random.normal(0, 3, len(dates)),
    "MSFT": np.linspace(170, 135, len(dates)) + np.random.normal(0, 5, len(dates)),
    "GOOGL": np.linspace(70, 50, len(dates)) + np.random.normal(0, 2, len(dates)),
    "AMZN": np.linspace(95, 80, len(dates)) + np.random.normal(0, 4, len(dates)),
}
historical_crisis_prices = pd.DataFrame(mock_data, index=dates)

# 2. 当前投资组合配置
current_portfolio = {
    "AAPL": 1000,   # 持股数量
    "MSFT": 500,
    "GOOGL": 800,
    "AMZN": 400
}
# 假设当前的个股单价 (Current Spot Prices)
current_prices = pd.Series({"AAPL": 180.0, "MSFT": 420.0, "GOOGL": 175.0, "AMZN": 185.0})

# --- 核心量化步骤 ---

def run_historical_stress_test(crisis_df, current_prices, portfolio_qty):
    """
    运行历史场景压力测试
    :param crisis_df: 历史危机时期的价格 DataFrame
    :param current_prices: 当前资产的最新价格 Series
    :param portfolio_qty: 当前投资组合的持仓数量 dict
    """
    # 计算当前组合总市值
    qty_series = pd.Series(portfolio_qty)
    current_values = current_prices * qty_series
    initial_portfolio_value = current_values.sum()
    
    print("===== 当前投资组合状态 =====")
    for t in tickers:
        print(f"{t}: 持仓 {qty_series[t]} 股 | 当前单价: ${current_prices[t]:.2f} | 市值: ${current_values[t]:,.2f}")
    print(f"当前组合总价值: ${initial_portfolio_value:,.2f}\n")

    # 步骤 1: 计算历史危机期间的累积收益率 / 最大冲击幅度 (Peak-to-Trough)
    # 在历史场景法中，通常取该期间的【最大单日跌幅】或【整个周期的极端绝对跌幅】
    # 这里我们采用【场景区间内最大累计跌幅】来模拟“压力最重”的一天
    
    # 计算整个历史危机区间的资产变化率 (以区间初始价格为基准的极端变动)
    # 或者也可以直接计算该历史区间内 [最低价 / 初始价 - 1]
    crisis_returns = (crisis_df.min() - crisis_df.iloc[0]) / crisis_df.iloc[0]
    
    print("===== 提取的历史场景冲击因子 (Shock Factors) =====")
    for t, ret in crisis_returns.items():
        print(f"{t} 预计遭受冲击跌幅: {ret*100:.2f}%")
    print("")

    # 步骤 2: 将冲击因子应用到当前资产价格中，得到压力下的新价格
    stressed_prices = current_prices * (1 + crisis_returns)
    
    # 步骤 3: 重新评估资产价值 (Re-pricing)
    stressed_values = stressed_prices * qty_series
    stressed_portfolio_value = stressed_values.sum()
    
    # 步骤 4: 计算损益 (P&L) 和最大潜在损失率
    portfolio_pnl = stressed_portfolio_value - initial_portfolio_value
    loss_percentage = (portfolio_pnl / initial_portfolio_value) * 100
    
    # --- 结果展示 ---
    results_df = pd.DataFrame({
        "Current Price": current_prices,
        "Shock Factor (%)": crisis_returns * 100,
        "Stressed Price": stressed_prices,
        "Current Value": current_values,
        "Stressed Value": stressed_values,
        "Asset P&L": stressed_values - current_values
    })
    
    print("===== 压力测试详细结果 =====")
    print(results_df.round(2).to_string())
    print("-" * 50)
    print(f"压力下投资组合总价值变为: ${stressed_portfolio_value:,.2f}")
    print(f"预计绝对亏损 (P&L): ${portfolio_pnl:,.2f}")
    print(f"投资组合预期回撤幅度: {loss_percentage:.2f}%")
    
    return results_df

# 执行测试
results = run_historical_stress_test(historical_crisis_prices, current_prices, current_portfolio)
```
