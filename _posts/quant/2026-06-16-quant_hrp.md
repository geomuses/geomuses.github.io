# HRP?

**HRP (Hierarchical Risk Parity, 层级风险平配)**

一种革命性资产配置算法

# HRP 的降维打击原理：

HRP 完全**不需要对协方差矩阵求逆**。它引入了**机器学习中的聚类算法**（图论与层级聚类），把资产配置分成了三步：

1. **层级聚类（Hierarchical Clustering）**：根据资产之间的相关性，把它们像“家谱树（Dendrogram）”一样组织起来。基因相似的（如 `NVDA`, `GOOGL`, `QQQ` 等高科技）聚在左边分支，避险资产（如 `GLDM`, `USD`）聚在右边分支。
    
2. **矩阵准角形化（Matrix Diagonalization）**：把资产按照树状图的顺序重新排列，让相关的资产挨在一起，不相关的离得远。
    
3. **递归平分（Recursive Bisection）**：顺着树状图的分支，从上到下进行“风险平配（Risk Parity）”。先在“科技集群”和“避险集群”之间分配资金，然后再深入到科技集群内部，在 `NVDA` 和 `GOOGL` 之间分配资金。

这种“基因隔离”的方法带来了一个巨大好处：**哪怕危机来临、相关性飙升，由于各资产在树状图结构中早已被切断了联动，它的风险永远不会集中在某一个单一风险源上。**

```python
import numpy as np
import pandas as pd
import yfinance as yf
import matplotlib.pyplot as plt
from scipy.cluster.hierarchy import linkage
from scipy.spatial.distance import squareform

def get_ivp(covariance):
    """
    计算逆方差权重 (Inverse-Variance Portfolio)
    这是最基础的风险平配：资产波动越大，分配权重越小
    """
    variance = np.diag(covariance)
    ivp = 1.0 / variance
    ivp /= ivp.sum()
    return ivp

def get_cluster_var(covariance, cluster_indices):
    """
    计算某一个资产集群的整体内部方差
    """
    cluster_cov = covariance.iloc[cluster_indices, cluster_indices]
    w = get_ivp(cluster_cov)
    cluster_var = np.dot(np.dot(w, cluster_cov), w)
    return cluster_var

def get_quasi_diag(link):
    """
    矩阵准角形化：根据聚类树(linkage)，对资产进行重新排序
    让相互关联的资产在序列中排在一起
    """
    link = link.astype(int)
    sort_ix = pd.Series([link[-1, 0], link[-1, 1]])
    num_items = link[-1, 3]
    
    while sort_ix.max() >= num_items:
        sort_ix.index = range(0, sort_ix.shape[0] * 2, 2)
        df0 = sort_ix[sort_ix >= num_items]
        i = df0.index
        j = df0.values - num_items
        sort_ix[i] = link[j, 0]
        df0 = pd.Series(link[j, 1], index=i + 1)
        sort_ix = sort_ix.append(df0)
        sort_ix = sort_ix.sort_index()
        sort_ix.index = range(sort_ix.shape[0])
        
    return sort_ix.tolist()

def recursive_bisection(covariance, sort_ix):
    """
    递归平分算法：自上而下顺着树枝分配权重
    """
    weights = pd.Series(1.0, index=sort_ix)
    cluster_items = [sort_ix]  # 初始为包含所有资产的单集群
    
    while len(cluster_items) > 0:
        # 弹出当前需要分裂的集群
        cluster_items = [c[i:j] for c in cluster_items for i, j in ((0, len(c) // 2), (len(c) // 2, len(c))) if len(c) > 1]
        
        # 两两配对，进行风险权重划分
        for i in range(0, len(cluster_items), 2):
            cluster_1 = cluster_items[i]
            cluster_2 = cluster_items[i + 1]
            
            var_1 = get_cluster_var(covariance, cluster_1)
            var_2 = get_cluster_var(covariance, cluster_2)
            
            # 计算风险分配因子 (方差大的分配少，方差小的分配多)
            alpha = 1 - var_1 / (var_1 + var_2)
            
            # 更新左分支和右分支资产的权重
            weights[cluster_1] *= alpha
            weights[cluster_2] *= (1 - alpha)
            
    return weights

def hrp_portfolio(returns):
    """
    HRP 顶层调用函数
    """
    # 计算资产间相关系数矩阵与协方差矩阵
    corr = returns.corr()
    cov = returns.cov()
    
    # 将相关系数转化为“距离矩阵” (相关性越高，距离越近)
    dist = np.sqrt(2 * (1 - corr).clip(lower=0))
    dist_compressed = squareform(dist)
    
    # 步骤 1: 层级聚类 (单联动凝聚聚类)
    link = linkage(dist_compressed, method='single')
    
    # 步骤 2: 准角形化重新排序
    sort_ix = get_quasi_diag(link)
    sorted_tickers = corr.index[sort_ix].tolist()
    
    # 步骤 3: 递归平分计算权重
    weights = recursive_bisection(cov, sorted_tickers)
    return weights

# ==========================================
# 2. 实盘数据下载与运行演示
# ==========================================
if __name__ == "__main__":
    plt.style.use("ggplot")
    
    # 使用你刚才跑出肥尾数据的标的组合
    tickers = ["COHR", "GLDM", "GOOGL", "NVDA", "QQQ", "VOO", "USD"]
    
    print("正在下载资产历史数据...")
    data = yf.download(tickers, period="2y")['Close']
    log_returns = np.log(data / data.shift(1)).dropna()
    
    # 运行 HRP 权重计算
    hrp_weights = hrp_portfolio(log_returns)
    
    print("\n========= HRP 模型换代最终资产权重 =========")
    for ticker, weight in hrp_weights.items():
        print(f"资产 {ticker:<6} : 分配权重 {weight*100:>6.2f}%")
    print(f"权重总和: {hrp_weights.sum():.2f}")
    
    # 绘制权重分布图
    plt.figure(figsize=(10, 5))
    hrp_weights.plot(kind='bar', color='darkslateblue', alpha=0.8)
    plt.title("Hierarchical Risk Parity (HRP) Weights Allocation", fontsize=12, fontweight='bold')
    plt.ylabel("Weights")
    plt.xlabel("Assets")
    plt.xticks(rotation=45)
    plt.tight_layout()
    plt.show()
```
