# -*- coding: utf-8 -*-
import os
import random
import time
import numpy as np
import matplotlib.pyplot as plt
from tqdm import tqdm

# ==========================================
# 1. 配置 Keras 后端为 Torch (保持与原作业一致)
# ==========================================
os.environ["KERAS_BACKEND"] = "torch"
import keras
from keras import layers, ops

# ==========================================
# 2. 初始化 Gymnasium 环境 (本地弹窗模式)
# ==========================================
import gymnasium as gym

def make_lunar_lander_env(render_mode='rgb_array'):
    try:
        return gym.make('LunarLander-v2', render_mode=render_mode)
    except gym.error.DeprecatedEnv:
        return gym.make('LunarLander-v3', render_mode=render_mode)

class GymCompatEnv:
    def __init__(self, env):
        self._env = env
        self.observation_space = env.observation_space
        self.action_space = env.action_space

    def reset(self):
        obs, _ = self._env.reset()
        return obs

    def step(self, action):
        obs, reward, terminated, truncated, info = self._env.step(action)
        return obs, reward, terminated or truncated, info

    def render(self, mode='rgb_array'):
        return self._env.render()

# ==========================================
# 3. 固定随机种子 (请勿修改)
# ==========================================
seed = 543 
def fix(env, seed):
    env.reset(seed=seed)
    env.action_space.seed(seed)
    keras.utils.set_random_seed(seed)
    np.random.seed(seed)
    random.seed(seed)

# ==========================================
# 4. 定义策略梯度网络与智能体
# ==========================================
class PolicyGradientNetwork(keras.Model):
    def __init__(self):
        super().__init__()
        # 恢复为经典的 128->64->4 架构，更适合小学习率在 1200 batch 内收敛
        self.fc1 = layers.Dense(128, activation='relu')
        self.fc2 = layers.Dense(64, activation='relu')
        self.fc3 = layers.Dense(4)

    def call(self, x, training=False):
        x = self.fc1(x)
        x = self.fc2(x)
        return ops.softmax(self.fc3(x), axis=-1)

gamma_constant = 0.999 

class PolicyGradientAgent():
    def __init__(self, network, gamma_discount=gamma_constant):
        self.network = network
        # 建议本地调大到 1e-3 或 5e-4 提升收敛速度，这里先保持 1e-4 兼顾稳定
        self.optimizer = keras.optimizers.Adam(learning_rate=1e-4)
        self.optimizer.build(self.network.trainable_weights)
        self.gamma_discount = gamma_discount
        self._training = True

    def train(self):
        self._training = True

    def eval(self):
        self._training = False

    def forward(self, state):
        state = ops.convert_to_tensor(state, dtype='float32')
        if len(ops.shape(state)) == 1:
            state = ops.expand_dims(state, 0)
        return self.network(state, training=self._training)

    def learn(self, log_probs, returns):
        # 清空梯度
        for weight in self.network.trainable_weights:
            if weight.value.grad is not None:
                weight.value.grad.zero_()
                
        log_probs = ops.stack(log_probs)
        returns = ops.convert_to_tensor(returns, dtype='float32')
        loss = -ops.sum(log_probs * returns)
        loss.backward()
        
        grads = [weight.value.grad for weight in self.network.trainable_weights]
        self.optimizer.apply_gradients(zip(grads, self.network.trainable_weights))

    def sample(self, state):
        state_tensor = ops.convert_to_tensor(state, dtype='float32')
        if len(ops.shape(state_tensor)) == 1:
            state_tensor = ops.expand_dims(state_tensor, 0)
        action_prob = self.network(state_tensor, training=self._training)
        action = int(np.random.choice(
            action_prob.shape[-1],
            p=ops.convert_to_numpy(action_prob[0]),
        ))
        log_prob = ops.log(action_prob[0, action] + 1e-8)
        return action, log_prob

# ==========================================
# 5. 训练阶段 (使用无画面的 rgb_array 模式加速)
# ==========================================
print("正在初始化训练环境...")
_train_raw_env = make_lunar_lander_env(render_mode='rgb_array')
fix(_train_raw_env, seed)
train_env = GymCompatEnv(_train_raw_env)

network = PolicyGradientNetwork()
# 哑前向传播以初始化网络参数
network(ops.convert_to_tensor(np.zeros((1, 8), dtype='float32')))
agent = PolicyGradientAgent(network, gamma_discount=gamma_constant)

agent.train()  
EPISODE_PER_BATCH = 5 
NUM_BATCH = 1200       
avg_total_rewards = []

print(f"开始训练智能体 (Total {NUM_BATCH} Batches)...")
prg_bar = tqdm(range(NUM_BATCH))
for batch in prg_bar:
    batch_log_probs, batch_returns = [], [] 
    total_rewards = []

    for episode in range(EPISODE_PER_BATCH):
        state = train_env.reset()
        episode_log_probs = [] 
        episode_immediate_rewards = [] 
        total_reward = 0

        while True:
            action, log_prob = agent.sample(state) 
            next_state, reward, done, _ = train_env.step(action)

            episode_log_probs.append(log_prob)
            episode_immediate_rewards.append(reward) 

            state = next_state
            total_reward += reward

            if done:
                returns_t = []
                G = 0
                for r in reversed(episode_immediate_rewards):
                    G = r + agent.gamma_discount * G 
                    returns_t.insert(0, G)

                batch_log_probs.extend(episode_log_probs)
                batch_returns.extend(returns_t) 
                total_rewards.append(total_reward)
                break

    avg_total_reward = sum(total_rewards) / len(total_rewards)
    avg_total_rewards.append(avg_total_reward)
    prg_bar.set_description(f"Batch {batch+1} - Avg Reward: {avg_total_reward: 4.1f}")

    # 归一化并更新网络
    batch_returns_np = np.array(batch_returns)
    batch_returns_normalized = (batch_returns_np - np.mean(batch_returns_np)) / (np.std(batch_returns_np) + 1e-9)
    agent.learn(batch_log_probs, batch_returns_normalized)

_train_raw_env.close()
print("训练完成！")

# 绘制训练曲线
plt.plot(avg_total_rewards)
plt.title("Training Total Rewards")
plt.xlabel("Batch")
plt.ylabel("Reward")
plt.show(block=False) # 不阻塞后续代码执行
plt.pause(2)
plt.close()

# ==========================================
# 6. 本地测试与可视化阶段 (弹出 Pygame 窗口)
# ==========================================
print("\n正在拉起本地图形窗口进行测试...")
_test_raw_env = make_lunar_lander_env(render_mode='human')
fix(_test_raw_env, seed)
test_env = GymCompatEnv(_test_raw_env)

agent.eval()  
NUM_OF_TEST = 5 
test_total_reward = []
action_list = []

for i in range(NUM_OF_TEST):
    actions = []
    state = test_env.reset()
    total_reward = 0
    done = False
    
    print(f"第 {i+1} 次测试开始...")
    while not done:
        action, _ = agent.sample(state)
        actions.append(action)
        
        # 在 render_mode='human' 下，step 会自动刷新本地 Pygame 窗口
        state, reward, done, _ = test_env.step(action)
        total_reward += reward
        
        # 控制动画帧率 (~30 FPS)，防止飞船闪现坠毁
        time.sleep(1.0 / 30.0) 
        
    print(f"第 {i+1} 次测试结束，总得分: {total_reward:.2f}")
    test_total_reward.append(total_reward)
    action_list.append(actions)

_test_raw_env.close()

# ==========================================
# 7. 保存结果 (用于提交)
# ==========================================
# PATH = "Action_List.npy"
# np.save(PATH, np.array(action_list, dtype=object))
print(f"\n全部测试完成! 平均得分: {np.mean(test_total_reward):.2f}")
