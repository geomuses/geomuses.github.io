```python
import time , os 
import numpy as np
import gymnasium as gym
from gymnasium import spaces
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.common.action_chains import ActionChains
from selenium.webdriver.firefox.service import Service
from selenium.webdriver.firefox.options import Options
from webdriver_manager.firefox import GeckoDriverManager

# 强制将 Selenium 和浏览器的临时目录指定到当前用户的家目录下
os.environ["TMPDIR"] = os.path.expanduser("~/tmp")
os.makedirs(os.environ["TMPDIR"], exist_ok=True)

class ChromeDinoEnv(gym.Env):
    """
    Chrome Dino 游戏自定义 Gym 环境
    """
    metadata = {"render_modes": ["human"], "render_fps": 30}

    def __init__(self, render_mode="human", headless=False):
        super(ChromeDinoEnv, self).__init__()

        # 定义动作空间: 0: 无操作, 1: 跳跃 (Up), 2: 蹲下 (Down)
        self.action_space = spaces.Discrete(3)

        # 定义观察空间: [当前速度, 障碍物xPos, 障碍物yPos, 障碍物width]
        # 使用 float32 兼容主流 RL 库
        self.observation_space = spaces.Box(
            low=np.array([0, 0, 0, 0]), 
            high=np.array([100, 1000, 500, 200]), 
            dtype=np.float32
        )

        # Selenium 配置
        options = Options()
        # Firefox 静音首选项（你之前注释掉的，如果是 Firefox 需要用 set_preference）
        options.set_preference("media.volume_scale", "0.0")

        #if headless:
         #   options.add_argument("-headless")
        if headless:
            # ...
            options.add_argument("--headless")
        # options.set_preference("media.volume_scale", "0.0")
        
        service = Service(GeckoDriverManager().install())
        self.driver = webdriver.Firefox(service=service, options=options)
        self.driver.set_window_size(1100, 650)
        self.url = "https://chromedino.com"
        
        self.body = None
        self._is_ducking = False
        self._load_game()

    def _load_game(self):
        self.driver.get(self.url)
        # 等待 Runner 加载
        while not self.driver.execute_script("return typeof Runner !== 'undefined'"):
            time.sleep(0.5)
        self.body = self.driver.find_element(By.TAG_NAME, "body")

    def _get_obs(self):
        """获取当前游戏状态"""
        # 获取最近的一个障碍物
        obs_data = self.driver.execute_script("""
            var runner = Runner.instance_;
            var obs = runner.horizon.obstacles[0];
            return {
                speed: runner.currentSpeed,
                xPos: obs ? obs.xPos : 1000,
                yPos: obs ? obs.yPos : 0,
                width: obs ? obs.width : 0
            };
        """)
        return np.array([
            obs_data['speed'],
            obs_data['xPos'],
            obs_data['yPos'],
            obs_data['width']
        ], dtype=np.float32)

    def reset(self, seed=None, options=None):
        """重置环境"""
        super().reset(seed=seed)
        
        # 如果已经结束，按空格重启；如果是首次启动，也按空格
        self.body.send_keys(Keys.SPACE)
        time.sleep(0.5) # 等待动画
        
        observation = self._get_obs()
        info = {}
        return observation, info

    def step(self, action):
        """执行动作并返回结果"""
        # 执行动作
        if action == 1: # Jump
            self.body.send_keys(Keys.ARROW_UP)
            if self._is_ducking: self._is_ducking = False
        elif action == 2: # Duck
            if not self._is_ducking:
                ActionChains(self.driver).key_down(Keys.ARROW_DOWN).perform()
                self._is_ducking = True
        else: # No-op or end ducking
            if self._is_ducking:
                ActionChains(self.driver).key_up(Keys.ARROW_DOWN).perform()
                self._is_ducking = False

        # 获取状态
        obs = self._get_obs()
        
        # 检测是否碰撞
        terminated = self.driver.execute_script("return Runner.instance_.crashed;")
        
        # 奖励机制
        # 存活每一帧给 0.1 奖励，碰撞扣 10 分
        reward = 0.1 if not terminated else -10.0
        
        truncated = False
        # info = {"score": self.driver.execute_script("return Runner.instance_.distanceMeter.digits").join("")}
        info = {"score": "".join(self.driver.execute_script("return Runner.instance_.distanceMeter.digits"))}
        return obs, reward, terminated, truncated, info

    def render(self):
        # 浏览器本身就是渲染窗口
        pass

    def close(self):
        if self.driver:
            self.driver.quit()

# --- 测试运行 ---
if __name__ == "__main__":

    env = ChromeDinoEnv(headless=False)
    
    for episode in range(3):
        obs, info = env.reset()
        done = False
        total_reward = 0
        
        print(f"开始第 {episode + 1} 局游戏...")
        
        while not done:
            # 这里可以用你的 AI 逻辑代替随机动作
            # 示例：简单的阈值判断
            if obs[1] < 150: # 如果障碍物 xPos < 150
                action = 1   # 跳
            else:
                action = 0   # 不动
                
            obs, reward, terminated, truncated, info = env.step(action)
            total_reward += reward
            
            if terminated or truncated:
                done = True
                print(f"游戏结束，总奖励: {total_reward:.2f}")
                time.sleep(1)

    env.close()
```

定义一个继承自 `gym.Env` 的自定义环境类。所有标准 RL 环境都必须继承这个基类，并实现 `reset()`、`step()`、`close()` 方法。

# `_get_obs` 方法

```python
def _get_obs(self):
```

私有方法，从游戏引擎中提取当前状态作为观察值。

```python
    obs_data = self.driver.execute_script("""
        var runner = Runner.instance_;
        var obs = runner.horizon.obstacles[0];
        return {
            speed: runner.currentSpeed,
            xPos: obs ? obs.xPos : 1000,
            yPos: obs ? obs.yPos : 0,
            width: obs ? obs.width : 0
        };
    """)
```

在浏览器内执行 JavaScript：

- `Runner.instance_`：获取游戏单例对象
- `runner.horizon.obstacles[0]`：获取最近的第一个障碍物
- 若无障碍物，`xPos` 默认返回 1000（表示很远），其余为 0
- 返回速度、障碍物坐标和宽度的字典

```python
    return np.array([
        obs_data['speed'],
        obs_data['xPos'],
        obs_data['yPos'],
        obs_data['width']
    ], dtype=np.float32)
```

将 Python 字典转换为 NumPy float32 数组，符合观察空间的格式要求。

# `step` 方法

```python
def step(self, action):
```

核心方法，接受一个动作，执行后返回新状态。

```python
    if action == 1:
        self.body.send_keys(Keys.ARROW_UP)
        if self._is_ducking: self._is_ducking = False
```

动作 1（跳跃）：发送上箭头键。如果之前在蹲下，更新状态标志为非蹲下。

```python
    elif action == 2:
        if not self._is_ducking:
            ActionChains(self.driver).key_down(Keys.ARROW_DOWN).perform()
            self._is_ducking = True
```

动作 2（蹲下）：只在未蹲下时才执行。用 `key_down`（持续按住）而非 `send_keys`（瞬间按键），模拟长按蹲下的效果。

```python
    else:
        if self._is_ducking:
            ActionChains(self.driver).key_up(Keys.ARROW_DOWN).perform()
            self._is_ducking = False
```

动作 0（无操作）：如果当前在蹲下，则释放下箭头键，恢复站立姿势。

```python
    obs = self._get_obs()
```

执行动作后立即获取新的游戏状态。

```python
    terminated = self.driver.execute_script("return Runner.instance_.crashed;")
```

通过 JavaScript 查询游戏的 `crashed` 属性，判断恐龙是否撞上障碍物（即 episode 是否结束）。

```python
    reward = 0.1 if not terminated else -10.0
```

奖励设计：存活每一步给 +0.1 奖励（鼓励活得久），碰撞给 -10 惩罚（强烈惩罚死亡）。

```python
    truncated = False
```

`truncated` 表示因时间限制等外部原因强制截断（区别于自然终止）。此环境无时间限制，始终为 False。

```python
    info = {"score": self.driver.execute_script("return Runner.instance_.distanceMeter.digits").join("")}
```

从游戏中读取分数的各位数字数组，用 `join("")` 拼接成字符串（注意：这里有一个 bug——`.join()` 是 Python 字符串方法，应写成 `"".join(...)`，而非在 JS 返回值上直接调用）。

```python
    return obs, reward, terminated, truncated, info
```

返回标准的五元组，这是 Gymnasium `step()` 的规范返回格式。

# `render` 和 `close`

```python
def render(self):
    pass
```

渲染方法为空，因为浏览器本身就是可视化界面，无需额外渲染。

```python
def close(self):
    if self.driver:
        self.driver.quit()
```

关闭浏览器，释放资源。`if self.driver` 防止未初始化时调用出错。
