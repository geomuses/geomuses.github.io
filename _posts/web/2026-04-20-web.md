#  同步的爬虫脚本

```python
import requests
from bs4 import BeautifulSoup

def simple_wiki_spider(url):
    # 1. 发送请求
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"
    }
    response = requests.get(url, headers=headers)
    
    # 检查请求是否成功
    if response.status_code != 200:
        print("网页请求失败！")
        return

    # 2. 解析网页内容
    soup = BeautifulSoup(response.text, 'html.parser')

    # 获取页面主标题
    main_title = soup.find('h1').text
    print(f"--- 页面标题: {main_title} --- \n")

    # 获取第一段简介
    # 维基百科的简介通常在 content 里的第一个 <p> 标签
    first_paragraph = soup.select_one('#mw-content-text .mw-parser-output > p:not([class])')
    if first_paragraph:
        print(f"内容简介:\n{first_paragraph.text.strip()}\n")

    # 3. 获取所有二级标题 (H2)
    print("--- 目录大纲 ---")
    headlines = soup.find_all('h2')
    for i, h in enumerate(headlines[:10], 1):  # 只列出前10个
        title_text = h.text.replace('[edit]', '').strip()
        if title_text:
            print(f"{i}. {title_text}")

# 执行爬虫
target_url = "https://en.wikipedia.org/wiki/Python_(programming_language)"
simple_wiki_spider(target_url)
```

# 异步的爬虫脚本

```python
import asyncio
import aiohttp
from bs4 import BeautifulSoup

# 将爬取逻辑封装在异步函数中
async def fetch_wiki_data(session, url):
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"
    }
    print(f"正在异步请求: {url}")
    # 异步发送请求
    async with session.get(url, headers=headers) as response:
        if response.status != 200:
            print(f"请求失败: {url}")
            return
        # 异步读取响应内容
        html = await response.text()
        # 解析（BeautifulSoup 本身是同步的，但在异步流中运行很快，通常不需要专门异步化）
        soup = BeautifulSoup(html, 'html.parser')
        title = soup.find('h1').text
        print(f"--- 抓取完成: {title} ---")
        # 提取简介
        first_p = soup.select_one('#mw-content-text .mw-parser-output > p:not([class])')
        if first_p:
            print(f"简介预览: {first_p.text[:50]}...")

async def main():
    urls = [
        "https://en.wikipedia.org/wiki/Python_(programming_language)",
        "https://en.wikipedia.org/wiki/Asynchronous_I/O",
        "https://en.wikipedia.org/wiki/Event_loop"
    ]
    
    # 创建全局 session，复用 TCP 连接
    async with aiohttp.ClientSession() as session:
        # 创建任务列表
        tasks = [fetch_wiki_data(session, url) for url in urls]
        # 并发执行所有任务
        await asyncio.gather(*tasks)

if __name__ == "__main__":
    # 启动异步事件循环
    asyncio.run(main())
```

在同步代码中，`requests.get` 每次都会建立和关闭连接。在异步代码中，`async with aiohttp.ClientSession()` 像是一个**连接池**。它在 `main` 函数中开启一次，所有协程共享这个池子，效率极高

当程序运行到 `await response.text()` 时，Python 会检查数据是否已经传输完毕。如果没有，它会**放下当前任务**，去处理 `tasks` 列表里的下一个 URL。这就是异步爬虫能“同时”爬取多个页面的奥秘。

执行效率
- **同步结构**：请求 A (等待) -> 得到 A -> 请求 B (等待) -> 得到 B。
- **异步结构**：请求 A、B、C 同时发出 -> 谁先返回数据就先处理谁。
